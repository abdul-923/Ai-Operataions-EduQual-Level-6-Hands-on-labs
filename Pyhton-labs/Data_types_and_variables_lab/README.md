## Creating variables of diff data types

name = "Jhon"  #String
age = 22  #Integer
is_student = True #Bolean
height = 175.4  #Float


## Printing variables with  Data types

print("Name:", name, "Type", type(name))
print("Age:", age, "Type:", type(age))
print("Is Student:", is_student, "Type:", type(is_student))
print("Height", height, "Type", type(height))

#OUTPUT:
Name: Jhon Type <class 'str'>
Age: 22 Type: <class 'int'>
Is Student: True Type: <class 'bool'>
Height 175.4 Type <class 'float'>
